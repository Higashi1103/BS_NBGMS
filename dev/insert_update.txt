	<script>
		$(function() {
			var regName = /^20\d{2}(0[1-9]|1[1-2])$/
			var regContents = /^\w+(,\s\w+)*$/

			$('#name').blur(function() {
				$('#nameError').text('');
					if ($('#name').val().trim() === '') {
						$('#nameError').text('クラス名前を入力してください').css('color','red');
						return;
					}
					if (!regName.test($('#name').val().trim())) {
						$('#nameError').text('6桁の数字を入力してください（例：20YYMM）').css('color', 'red');
						} else {
							$('#nameError').text('✓').css('color', 'green');
						}
					})

			$('#contents').blur(function() {
				$('#contentsError').text('');
					if ($('#contents').val().trim() === '') {
						$('#contentsError').text('コンテンツを入力してください').css('color', 'red');
							return;
							}
					if (!regContents.test($('#contents').val().trim())) {
						$('#contentsError').text('正確な格式で入力してください（例：Java, Python）').css('color', 'red');
					} else {
						$('#contentsError').text('✓').css('color','green');
						}
				})

			$('#instructor').blur(function() {
				$('#instructorError').text('');
					if ($('#instructor').val().trim() === '') {
						$('#instructorError').text('講師の名前を入力してください').css('color', 'red');
							return;
						}
					if ($('#instructor').val().length > 0) {
						$('#instructorError').text('✓').css('color','green');
						}
					})
			
			$('#updateForm').on('submit', function(e) {
				e.preventDefault();
				
				if ($('#nameError').text().length > 1
					|| $('#contentsError').text().length > 1
					|| $('#instructorError').text().length > 1) {
				alert('入力項目を確認してください');
				} else {
					const $form = $(this);
					
					$.ajax({
						url:$form.attr('action'),
						type:$form.attr('method') || 'post',
						data:$form.serialize(),
						success:function() {
							$('#panel-body1').html(
								`<div style="text-align:center">
									<p>更新しました</p>
									<input type='button' value='戻る' onclick="location.href='/kurasu'" class="btn btn-warning">
								</div>
							`)
						},
						error:function() {
							alert('更新できませんでした')
						}
					})
				}
			})
			
			$('#insertForm').on('submit', function(e) {
				e.preventDefault();
				
				if ($('#name').val() == "" 
					|| $('#contents').val() == "" 
					|| $('#instructor').val() == ""
					|| $('#nameError').text().length > 1
					|| $('#contentsError').text().length > 1
					|| $('#instructorError').text().length > 1) {
				alert('入力項目を確認してください');
				} else {
					const $form = $(this);
					
					$.ajax({
						url:$form.attr('action'),
						type:$form.attr('method') || 'post',
						data:$form.serialize(),
						success:function() {
							$('#panel-body2').html(
								`<div style="text-align:center">
									<p>登録しました</p>
									<input type='button' value='戻る' onclick="location.href='/kurasu'" class="btn btn-warning">
								</div>				
							`)
						},
						error:function() {
							alert('登録できませんでした')
						}
					})
				}
			})
		})
	</script>
