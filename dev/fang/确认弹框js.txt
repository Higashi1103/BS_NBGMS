HenkoConfirm.js

$(function() {

	$("#openConfirmModal").on("click", function() {
		 
		let namae = $("#namaeInput").val();
        let kagen = $("input[name='kagenTensu']").val();
        let riyuValue = $("#gutaiRiyuSelect").val(); 
        let riyuText = $("select[name='gutaiRiyu'] option:selected").text(); 
        let hassei = $("input[name='hasseiBi']").val();
        let bikou = $("input[name='bikou']").val();
        
        $("#riyuIdInput").val(riyuValue);

		$("#confirmNamae").text(namae || "（未入力）");
        $("#confirmKagen").text(kagen || "（未入力）");
        $("#confirmRiyu").text(riyuText || "（未選択）"); 
        $("#confirmHassei").text(hassei || "（未入力）");
        $("#confirmBikou").text(bikou || "（未入力）");
        
        $("#confirmModal").modal("show");

	});

	$("#confirmSubmit").on("click", function() {
		$("#updateForm").submit();
	});
});
